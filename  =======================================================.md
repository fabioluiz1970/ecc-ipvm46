const load = (k, d) => JSON.parse(localStorage.getItem(k)) || d;  
const save = (k, v) => localStorage.setItem(k, JSON.stringify(v));  
  
let params = load("ECC_PARAMS", {  
    locais: [], equipes: [], atividades: [], acoes: []  
});  
let cronograma = load("ECC_CRONO", []);

======================================================== */  
function openTab(tab){  
    document.querySelectorAll(".tab").forEach(t => t.classList.remove("active"));  
    document.getElementById(tab).classList.add("active");  
}

function maskHora(i){  
    let v = i.value.replace(/\D/g, "").slice(0,4);  
    if(v.length > 2) i.value = v.slice(0,2) + ":" + v.slice(2);  
    else i.value = v;  
}  
  
/* =======================================================  
   CADASTRO BASE (INCLUIR / EDITAR / EXCLUIR)  
======================================================== */  
function addBase(tipo, inputId){  
    const val = document.getElementById(inputId).value.trim();  
    if(!val) return;  
    if(!params[tipo].includes(val)){  
        params[tipo].push(val);  
        save("ECC_PARAMS", params);  
    }  
    document.getElementById(inputId).value = "";  
    renderCadastros();  
}  
  
function editBase(tipo, index){  
    const novo = prompt("Alterar:", params[tipo][index]);  
    if(novo && novo.trim()){  
        params[tipo][index] = novo.trim();  
        save("ECC_PARAMS", params);  
        renderCadastros();  
    }  
}  
  
function delBase(tipo, index){  
    if(confirm("Excluir?")){  
        params[tipo].splice(index, 1);  
        save("ECC_PARAMS", params);  
        renderCadastros();  
    }  
}

function calcularTempo(ini, fim){  
    if(!ini || !fim) return "0h00m";  
    let [ih, im] = ini.split(":").map(Number);  
    let [fh, fm] = fim.split(":").map(Number);  
    let tot = (fh*60 + fm) - (ih*60 + im);  
    if(tot < 0) tot += 1440;  
    return `{(tot%60).toString().padStart(2,"0")}m`;  
}  
  
document.getElementById("cronogramaForm").addEventListener("submit", e => {  
    e.preventDefault();  
  
    let id = document.getElementById("crono-id").value;  
    let obj = {  
        dia: document.getElementById("crono-dia").value,  
        atividade: document.getElementById("crono-atividade").value,  
        equipe: document.getElementById("crono-equipe").value,  
        local: document.getElementById("crono-local").value,  
        inicioPrev: document.getElementById("crono-inicio").value,  
        fimPrev: document.getElementById("crono-fim").value,  
        tempoPrev: calcularTempo(  
            document.getElementById("crono-inicio").value,  
            document.getElementById("crono-fim").value  
        ),  
        obs: document.getElementById("crono-obs").value,  
        inicioReal:"", fimReal:""  
    };  
  
    if(id){  
        let idx = cronograma.findIndex(x => x.id == id);  
        obj.id = Number(id);  
        obj.numero = cronograma[idx].numero;  
        obj.inicioReal = cronograma[idx].inicioReal;  
        obj.fimReal = cronograma[idx].fimReal;  
        cronograma[idx] = obj;  
        document.getElementById("btnSalvarCrono").innerText = "Incluir";  
        document.getElementById("crono-id").value = "";  
    } else {  
        obj.id = Date.now();  
        obj.numero = cronograma.length + 1;  
        cronograma.push(obj);  
    }  
  
    save("ECC_CRONO", cronograma);  
    e.target.reset();  
    renderCronograma();  
    renderExecucao();  
    renderResumo();  
    renderTimeline();  
});  
  
function editCrono(id){  
    const item = cronograma.find(x=>x.id===id);  
    document.getElementById("crono-id").value = item.id;  
    document.getElementById("crono-dia").value = item.dia;  
    document.getElementById("crono-atividade").value = item.atividade;  
    document.getElementById("crono-equipe").value = item.equipe;  
    document.getElementById("crono-local").value = item.local;  
    document.getElementById("crono-inicio").value = item.inicioPrev;  
    document.getElementById("crono-fim").value = item.fimPrev;  
    document.getElementById("crono-obs").value = item.obs;  
    document.getElementById("btnSalvarCrono").innerText = "Salvar";  
}  
  
function delCrono(id){  
    if(confirm("Excluir linha?")){  
        cronograma = cronograma.filter(x=>x.id!==id);  
        cronograma.forEach((x,i)=> x.numero = i+1);  
        save("ECC_CRONO", cronograma);  
        renderCronograma();  
        renderExecucao();  
        renderResumo();  
        renderTimeline();  
    }  
}  
  
/* =======================================================  
   EXECUÇÃO — STATUS + TEMPO  
======================================================== */  
function mins(v){  
    if(!v) return null;  
    let [h,m] = v.split(":").map(Number);  
    return h*60+m;  
}  
  
function statusExec(item){  
    let ip = mins(item.inicioPrev);  
    let fp = mins(item.fimPrev);  
    let ir = mins(item.inicioReal);  
    let fr = mins(item.fimReal);  
    let now = new Date(), n = now.getHours()*60+now.getMinutes();  
  
    if(!ir) return {c:"st-nao", t:"Não iniciado"};  
  
    if(ir && !fr) return {c:"st-and", t:"Em andamento"};  
  
    if(fr <= fp) return {c:"st-ok", t:"Concluído"};  
  
    return {c:"st-at", t:"Atrasado"};  
}  
  
function tempoExec(item){  
    let ip = mins(item.inicioPrev);  
    let fp = mins(item.fimPrev);  
    let ir = mins(item.inicioReal);  
    let fr = mins(item.fimReal);  
    let now = mins(new Date().getHours()+":"+new Date().getMinutes());  
  
    if(!ir){  
        if(now < ip) return `Inicia em ${ip-now} min`;  
        return `Atrasado ${now-ip} min`;  
    }  
  
    if(ir && !fr){  
        let durPrev = fp - ir;  
        let durReal = now - ir;  
        let diff = durPrev - durReal;  
        if(diff >=0) return `Faltam ${diff} min`;  
        return `${Math.abs(diff)} min de atraso`;  
    }  
  
    let final = fr - fp;  
    if(final < 0) return `${Math.abs(final)} min adiantado`;  
    if(final > 0) return `${final} min atrasado`;  
    return "No horário";  
}  
  
function updateReal(id, campo, val){  
    let obj = cronograma.find(x=>x.id===id);  
    if(obj){  
        obj[campo] = val;  
        save("ECC_CRONO", cronograma);  
        renderExecucao();  
        renderResumo();  
        renderTimeline();  
    }  
}

function renderCadastros(){  
    const grupos = {  
        locais:"repLocais",  
        equipes:"repEquipes",  
        atividades:"repAtividades",  
        acoes:"repAcoes"  
    };  
  
    for(let t in grupos){  
        let box = document.getElementById(grupos[t]);  
        box.innerHTML = "";  
        params[t].forEach((txt,i)=>{  
            box.innerHTML += `  
                <div class="rep-line">  
                    <span onclick="editBase('{i})">${txt}</span>  
                    <div class="rep-actions">  
                        <button class="rep-edit" onclick="editBase('{i})">✎</button>  
                        <button class="rep-del" onclick="delBase('{i})">🗑</button>  
                    </div>  
                </div>  
            `;  
        });  
    }  
  
    /* atualizar selects */  
    document.getElementById("crono-atividade").innerHTML =  
        params.atividades.map(a=>`<option>${a}</option>`).join("");  
  
    document.getElementById("crono-equipe").innerHTML =  
        params.equipes.map(a=>`<option>${a}</option>`).join("");  
  
    document.getElementById("crono-local").innerHTML =  
        params.locais.map(a=>`<option>${a}</option>`).join("");  
}  
  
function renderCronograma(){  
    let tbody = document.getElementById("lista-cronograma");  
    tbody.innerHTML = "";  
  
    cronograma.forEach(item=>{  
        tbody.innerHTML += `  
            <tr>  
                <td>${item.numero}</td>  
                <td>${item.dia}</td>  
                <td>${item.atividade}</td>  
                <td>${item.equipe}</td>  
                <td>${item.local}</td>  
                <td>${item.inicioPrev}</td>  
                <td>${item.fimPrev}</td>  
                <td class="tempo">${item.tempoPrev}</td>  
                <td>${item.obs || "-"}</td>  
                <td>  
                    <button class="btn-edit" onclick="editCrono(${item.id})">✎</button>  
                    <button class="btn-delete" onclick="delCrono(${item.id})">🗑</button>  
                </td>  
            </tr>  
        `;  
    });  
}

            <p><strong>Previsto:</strong> {item.fimPrev} (${item.tempoPrev})</p>  
            <p><strong>Status:</strong> <span class="status {st.t}</span></p>  
            <p><strong>Tempo:</strong> ${tempoExec(item)}</p>  
            <p><strong>Obs:</strong> ${item.obs || "-"}</p>

/* =======================================================

function renderResumo(){  
    let tot = cronograma.length;  
    let ok  = cronograma.filter(x=>statusExec(x).c==="st-ok").length;  
    let and = cronograma.filter(x=>statusExec(x).c==="st-and").length;  
    let at  = cronograma.filter(x=>statusExec(x).c==="st-at").length;  
  
    let box = document.getElementById("blocoResumo");  
    box.innerHTML = `  
        <div class="resumo-card"><h3>Total</h3><div class="num">${tot}</div></div>  
        <div class="resumo-card"><h3>Concluídas</h3><div class="num">${ok}</div></div>  
        <div class="resumo-card"><h3>Andamento</h3><div class="num">${and}</div></div>  
        <div class="resumo-card"><h3>Atrasadas</h3><div class="num">${at}</div></div>  
    `;  
}  
  
/* =======================================================  
   TIMELINE  
======================================================== */  
function renderTimeline(){  
    let dia = document.getElementById("timelineDia").value;  
    let box = document.getElementById("timelineBox");  
    box.innerHTML = "";  
  
    cronograma.filter(x=>x.dia===dia)  
    .forEach(item=>{  
        let st = statusExec(item);  
        let info = tempoExec(item);  
  
        let i = mins(item.inicioPrev);  
        let f = mins(item.fimPrev);  
        let dur = f - i;  
        if(dur < 0) dur += 1440;  
  
        let px = Math.max(dur * 2, 50);  
  
        box.innerHTML += `  
            <div class="timeline-item">  
                <div class="timeline-time">{item.fimPrev}</div>  
                <div class="timeline-bar-container">  
                    <div class="timeline-bar {px}px">  
                        {item.atividade} (${info})  
                    </div>  
                </div>  
            </div>  
        `;  
    });  
}  
  
/* =======================================================

function exportarECC(){  
    let dados = {params, cronograma};  
    let blob = new Blob([JSON.stringify(dados,null,2)], {type:"application/json"});  
    let url = URL.createObjectURL(blob);  
    let a = document.createElement("a");  
    a.href = url;  
    a.download = "ECC_BACKUP.json";  
    a.click();  
}  
  
function importarECC(e){  
    const file = e.target.files[0];  
    if(!file) return;  
    const r = new FileReader();  
    r.onload = evt =>{  
        try{  
            let obj = JSON.parse(evt.target.result);  
            params = obj.params;  
            cronograma = obj.cronograma;  
            save("ECC_PARAMS", params);  
            save("ECC_CRONO", cronograma);  
            renderAll();  
            alert("Dados importados!");  
        }catch{  
            alert("Arquivo inválido.");  
        }  
    };  
    r.readAsText(file);  
}
